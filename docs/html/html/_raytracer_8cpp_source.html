47  const Vector3f &rayOrigin,
48  const Vector3f &rayDirection,
49  const std::vector<Sphere> &spheres,
50  const int &depth)
51 {
52  /* initialize */
53  float tnear = INFINITY;
54  const Sphere* sphere = NULL;
55 
56  /* find intersection of this ray with the sphere in the scene */
57  for (unsigned i = 0; i < spheres.size(); ++i) {
58  float t0 = INFINITY;
59  float t1 = INFINITY;
60  if (spheres[i].intersect(rayOrigin, rayDirection, t0, t1)) {
61  if (t0 < 0) t0 = t1;
62  if (t0 < tnear)
63  {
64  tnear = t0;
65  sphere = &spheres[i];
66  }
67  }
68  }
69 
70 
71  /* return black or background color if there is no intersection */
72  if (!sphere)
73  {
74  return Vector3f(0.3f);
75  }
76 
77  Vector3f surfaceColor = 0; /* intialize color of the ray/surfaceof the object intersected by the ray */
78  Vector3f pointOfIntersection = rayOrigin + rayDirection * tnear; /* point of intersection */
79  Vector3f normalOfIntersection = pointOfIntersection - sphere->position; /* normal at the intersection */
80  normalOfIntersection.normalize(); /* normalize */
81 
82  /* Start the checks and headaches */
83 
84  /* If the normal and the view direction are not opposite to each other, reverse the normal direction.
85  That means we are inside the sphere... so set the inside bool to true. Finally reverse the sign
86  of IdotN which we want positive. */
87 
88  float bias = 1e-4; /* add some bias to the point from which we will be tracing to retain some color */
89  bool inside = false; /* initalize */
90  if (rayDirection.dot(normalOfIntersection) > 0)
91  {
92  normalOfIntersection = -normalOfIntersection, inside = true;
93  }
94 
95  if ((sphere->transparency > 0 || sphere->reflection > 0) && depth < MAX_RAY_DEPTH)
96  {
97  float facingRatio = -rayDirection.dot(normalOfIntersection);
98  /* change the color mixing value to maximize the effect */
99  float fresnelEffect = weightDistribution(pow(1 - facingRatio, 3), 1, 0.1); /* this equation is everywhere */
100 
101  /* since all of the vectors are already normalized, the reflection direction is computed */
102  Vector3f reflectionDirection = rayDirection - normalOfIntersection * 2 * rayDirection.dot(normalOfIntersection);
103  reflectionDirection.normalize(); /* normalize */
104  Vector3f reflection = trace(pointOfIntersection + normalOfIntersection * bias, reflectionDirection, spheres, depth + 1);
105  Vector3f refraction = 0;
106 
107  /* if the sphere is also transparent compute refraction ray (transmission) */
108  if (sphere->transparency)
109  {
110  float ior = 1.1, eta = (inside) ? ior : 1 / ior; // are we inside or outside the surface?
111  float co = -normalOfIntersection.dot(rayDirection);
112  float k = 1 - eta * eta * (1 - co * co);
113  Vector3f refractionDirection = rayDirection * eta + normalOfIntersection * (eta * co - sqrt(k));
114  refractionDirection.normalize(); /* normalize */
115  refraction = trace(pointOfIntersection - normalOfIntersection * bias, refractionDirection, spheres, depth + 1);
116  }
117  /* the result is a weighted color distrobution value of reflection and refraction (if the sphere is transparent) */
118  surfaceColor = (reflection * fresnelEffect + refraction * (1 - fresnelEffect) * sphere->transparency) * sphere->surfaceColor;
119  }
120  else
121  {
122  /* a diffuse object, we can stop marching */
123  for (unsigned i = 0; i < spheres.size(); ++i)
124  {
125  if (spheres[i].emissionColor.x > 0)
126  {
127  // this is a light
128  Vector3f transmission = 1;
129  Vector3f lightDirection = spheres[i].position - pointOfIntersection;
130  lightDirection.normalize(); /* normalize */
131  for (unsigned j = 0; j < spheres.size(); ++j)
132  {
133  if (i != j)
134  {
135  float t0, t1;
136  if (spheres[j].intersect(pointOfIntersection + normalOfIntersection * bias, lightDirection, t0, t1))
137  {
138  transmission = 0;
139  break;
140  }
141  }
142  }
143  surfaceColor += sphere->surfaceColor * transmission * std::max(float(0), normalOfIntersection.dot(lightDirection)) * spheres[i].emissionColor;
144  }
145  }
146  }
147 
148  return surfaceColor + sphere->emissionColor;
149 }
